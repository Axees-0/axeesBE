glue_job_daily_sales.py
import sys from awsglue.transforms import * from awsglue.utils import getResolvedOptions from pyspark.context import SparkContext from awsglue.context import GlueContext from awsglue.job import Job from pyspark.sql.functions import col, when, to_timestamp from datetime import datetime # Initialize Glue context args = getResolvedOptions(sys.argv, ['JOB_NAME']) sc = SparkContext() glueContext = GlueContext(sc) spark = glueContext.spark_session job = Job(glueContext) job.init(args['JOB_NAME'], args) # Read data from S3 datasource0 = glueContext.create_dynamic_frame.from_options( connection_type="s3", connection_options={ "paths": ["s3://member-data-pipeline/raw/sales/"], "recurse": True }, format="csv", format_options={ "withHeader": True, "separator": "," } ) # Convert to DataFrame for complex transformations df = datasource0.toDF() # Apply business logic transformations # 1. Clean null values from critical fields df = df.dropna(subset=['customer_id', 'sale_amount', 'transaction_date']) # 2. Convert date strings to proper datetime df = df.withColumn('transaction_date', to_timestamp(col('transaction_date'), 'yyyy-MM-dd')) # 3. Apply sales categorization rules df = df.withColumn('sale_category', when(col('sale_amount') > 1000, 'High') .when(col('sale_amount') > 100, 'Medium') .otherwise('Low') ) # 4. Calculate regional tax based on state tax_rates = {'CA': 0.0725, 'NY': 0.08, 'TX': 0.0625} def calculate_tax(state, amount): return amount * tax_rates.get(state, 0.05) # Register UDF for tax calculation from pyspark.sql.functions import udf from pyspark.sql.types import DoubleType calculate_tax_udf = udf(calculate_tax, DoubleType()) df = df.withColumn('tax_amount', calculate_tax_udf(col('state'), col('sale_amount'))) # 5. Add processing timestamp df = df.withColumn('processed_at', lit(datetime.now())) # Convert back to DynamicFrame dynamic_frame_write = DynamicFrame.fromDF(df, glueContext, "dynamic_frame_write") # Write to S3 in Parquet format glueContext.write_dynamic_frame.from_options( frame=dynamic_frame_write, connection_type="s3", connection_options={ "path": "s3://member-data-pipeline/processed/sales/", "partitionKeys": ["year", "month", "day"] }, format="parquet", format_options={ "compression": "snappy" } ) job.commit()
